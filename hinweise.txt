### CONTAINER

### Git Pull
cd /opt/iobroker && npm i git+https://github.com/hammerappsde/iobroker.ws-bridge.git#main

### Adapter Update
cd /opt/iobroker && ./iobroker upload ws-bridge && ./iobroker restart ws-bridge

### Adapter Installieren
cd /opt/iobroker && ./iobroker add ws-bridge

### LÖSCHEN
./iobroker stop ws-bridge.0
./iobroker del ws-bridge.0
./iobroker del ws-bridge || true
./iobroker object del system.adapter.ws-bridge.0 --force || true
npm remove iobroker.ws-bridge || true
